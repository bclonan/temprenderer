function toSidebarItems(nodes: any[] | undefined, inheritedIcon?: string): SidebarMenuItem[] {
    const safe = Array.isArray(nodes) ? nodes : []

    return safe.map((n) => {
        const title = (n?.["@title"] ?? "").toString().trim()
        const level = (n?.["@menuLevel"] ?? "").toString().trim()

        const id = level || title || crypto.randomUUID()

        const rawUrl = (n?.["@url"] ?? "").toString().trim()
        const url = rawUrl.length ? rawUrl : undefined

        // âœ… icon by menuLevel, optional parent icon inheritance
        const icon = (level && FA_ICON_BY_MENU_LEVEL[level]) || inheritedIcon

        const children = toSidebarItems(n?.siteMapNode, icon)

        return {
            id,
            title: title || "(untitled)",
            url,
            icon,
            children,
        }
    })
}


// end function 

// new getter 
sidebarMenuItems: (s): SidebarMenuItem[] => {
    const top = s.model?.menu?.siteMapNode?.siteMapNode
    return toSidebarItems(top)
},



// new script in AppSiddebar
<script setup lang="ts">
import { reactive, computed } from "vue"
import { storeToRefs } from "pinia"
import { useRouter } from "vue-router"
import { useUserInformationStore } from "@/stores/user-information-store"

const router = useRouter()
const store = useUserInformationStore()

// If storeToRefs doesn't expose getters as refs in your setup, use computed fallback.
const sidebarMenuItems = computed(() => store.sidebarMenuItems)

const openMap = reactive<Record<string, boolean>>({})

function toggle(id: string) {
  openMap[id] = !openMap[id]
}

function navigate(item: any) {
  if (item.children?.length) {
    toggle(item.id)
    return
  }
  if (!item.url) return

  const isAbsolute = /^https?:\/\//i.test(item.url)
  if (isAbsolute) {
    window.location.href = item.url
    return
  }

  router.push(item.url)
}
</script>


// new template section in AppSidebar
<template>
  <b-sidebar shadow title="Menu" backdrop>
    <b-nav vertical>
      <template v-for="item in sidebarMenuItems" :key="item.id">
        <!-- TOP -->
        <div
          class="d-flex align-items-center justify-content-between px-3 py-2 sidebar-item"
          role="button"
          tabindex="0"
          @click="navigate(item)"
          @keydown.enter="navigate(item)"
        >
          <div class="d-flex align-items-center gap-2">
            <!-- Font Awesome icon -->
            <i v-if="item.icon" class="fa" :class="item.icon" />
            <span class="text-truncate">{{ item.title }}</span>
          </div>

          <!-- Toggle chevron (Font Awesome) -->
          <i
            v-if="item.children?.length"
            class="fa fa-chevron-down chevron"
            :class="{ open: !!openMap[item.id] }"
            @click.stop="toggle(item.id)"
          />

        <!-- CHILDREN -->

                  <span class="text-truncate">{{ child.title }}</span>
                </div>

                <i
                  v-if="child.children?.length"
                  class="fa fa-chevron-down chevron"

                </div>
              </b-collapse>
            </template>
          </div>
        </b-collapse>
      </template>
    </b-nav>
  </b-sidebar>
